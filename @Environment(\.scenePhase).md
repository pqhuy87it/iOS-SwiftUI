// Tương đương UIApplicationDelegate lifecycle methods trong UIKit:

// === 1a. Đọc scenePhase trong View ===

struct ScenePhaseDemo: View {
    @Environment(\.scenePhase) private var scenePhase
    @State private var lastPhase = "Unknown"
    
    var body: some View {
        VStack(spacing: 16) {
            Text("Scene Phase: \(lastPhase)")
                .font(.title2.bold())
            
            Circle()
                .fill(phaseColor)
                .frame(width: 60, height: 60)
                .overlay(
                    Image(systemName: phaseIcon)
                        .foregroundStyle(.white)
                        .font(.title3)
                )
            
            Text(phaseDescription)
                .font(.caption)
                .foregroundStyle(.secondary)
                .multilineTextAlignment(.center)
                .padding(.horizontal)
        }
        .onChange(of: scenePhase) { oldPhase, newPhase in
            lastPhase = "\(newPhase)"
            handlePhaseChange(from: oldPhase, to: newPhase)
        }
    }
    
    private var phaseColor: Color {
        switch scenePhase {
        case .active: return .green
        case .inactive: return .orange
        case .background: return .red
        @unknown default: return .gray
        }
    }
    
    private var phaseIcon: String {
        switch scenePhase {
        case .active: return "play.fill"
        case .inactive: return "pause.fill"
        case .background: return "moon.fill"
        @unknown default: return "questionmark"
        }
    }
    
    private var phaseDescription: String {
        switch scenePhase {
        case .active:
            return "App đang HIỂN THỊ và TƯƠNG TÁC được.\nUser đang dùng app."
        case .inactive:
            return "App hiển thị nhưng KHÔNG tương tác.\nĐang chuyển trạng thái (notification, app switcher)."
        case .background:
            return "App đã vào BACKGROUND.\nKhông hiển thị trên màn hình."
        @unknown default:
            return "Trạng thái không xác định."
        }
    }
    
    private func handlePhaseChange(from oldPhase: ScenePhase, to newPhase: ScenePhase) {
        switch newPhase {
        case .active:
            // App vừa trở lại foreground
            // → Refresh data, reconnect WebSocket, resume timers
            refreshDataIfNeeded()
            
        case .inactive:
            // App sắp bị che (notification banner, app switcher, Control Center)
            // → Tạm dừng operations nhạy cảm, blur sensitive data
            pauseSensitiveOperations()
            
        case .background:
            // App đã vào background
            // → Save data, cancel network requests, schedule background tasks
            saveAppState()
            
        @unknown default:
            break
        }
    }
    
    func refreshDataIfNeeded() { print("🔄 Refreshing data...") }
    func pauseSensitiveOperations() { print("⏸️ Pausing...") }
    func saveAppState() { print("💾 Saving state...") }
}

// App launch:     → .inactive → .active

// App Switcher:   .active → .inactive (vẫn inactive khi đang preview)
// Split View:     Cả 2 scenes có thể .active cùng lúc (iPad)

@main
struct MyApp: App {
    @Environment(\.scenePhase) private var scenePhase
    
    var body: some Scene {
        WindowGroup {
            ContentView()
        }
        .onChange(of: scenePhase) { _, newPhase in
            // Ở APP level: nhận phase changes cho TẤT CẢ scenes
            switch newPhase {
            case .background:
                // Schedule background refresh
                scheduleBackgroundRefresh()
            case .active:
                // Clear notification badges
                clearNotificationBadge()
            default:
                break
            }
        }
    }
    
    func scheduleBackgroundRefresh() { }
    func clearNotificationBadge() { }
}

// === 1c. Production patterns với scenePhase ===

@Observable
final class AppLifecycleManager {
    var isAppActive = true
    var lastBackgroundDate: Date?
    var shouldRefreshOnReturn = false
    
    func handlePhaseChange(_ phase: ScenePhase) {
        switch phase {
        case .active:
            isAppActive = true
            // Refresh nếu đã background > 5 phút
            if let lastBg = lastBackgroundDate,
               Date.now.timeIntervalSince(lastBg) > 300 {
                shouldRefreshOnReturn = true
            }
            
        case .inactive:
            // Blur sensitive data (banking app)
            // Pause video/audio
            break
            
        case .background:
            isAppActive = false
            lastBackgroundDate = .now
            // Persist state
            // Cancel unnecessary network requests
            // Stop location updates
            
        @unknown default:
            break
        }
    }
}

    // .leftToRight: English, Vietnamese, etc.

    // === 4d. dynamicTypeSize — Accessibility text size ===

    // .accessibility1 ... .accessibility5 (5 accessibility sizes)

        VStack(alignment: .leading, spacing: 16) {
            // Responsive layout based on size class
            Text("H: \(hSizeClass == .compact ? "Compact" : "Regular") | V: \(vSizeClass == .compact ? "Compact" : "Regular")")
                .font(.caption.monospaced())
            
            // Adaptive layout
            if hSizeClass == .compact {
                // iPhone: vertical layout
                VStack(alignment: .leading, spacing: 12) {
                    infoCard("Card 1", color: .blue)
                    infoCard("Card 2", color: .green)
                }
            } else {
                // iPad: horizontal layout
                HStack(spacing: 16) {
                    infoCard("Card 1", color: .blue)
                    infoCard("Card 2", color: .green)
                }
            }
            
            // Dynamic type awareness
            Text("Dynamic Type: \(typeSize.isAccessibilitySize ? "♿ Accessibility" : "Normal")")
                .font(.caption)
            
            // Layout direction
            Text("Direction: \(layoutDirection == .leftToRight ? "LTR →" : "← RTL")")
                .font(.caption)
            
            // Adapt for large text
            if typeSize >= .accessibility1 {
                Text("⚠️ Large text mode — UI đã adapt")
                    .font(.caption)
                    .foregroundStyle(.orange)
            }
        }
        .padding()
    }
    
    func infoCard(_ title: String, color: Color) -> some View {
        Text(title)
            .frame(maxWidth: .infinity)
            .padding()
            .background(color.opacity(0.15), in: .rect(cornerRadius: 12))
    }

// === 11a. Adaptive Layout Component ===

struct AdaptiveStack<Content: View>: View {
    @Environment(\.horizontalSizeClass) private var hSizeClass
    @Environment(\.dynamicTypeSize) private var typeSize
    
    let hSpacing: CGFloat
    let vSpacing: CGFloat
    @ViewBuilder let content: () -> Content
    
    init(hSpacing: CGFloat = 16, vSpacing: CGFloat = 12,
         @ViewBuilder content: @escaping () -> Content) {
        self.hSpacing = hSpacing
        self.vSpacing = vSpacing
        self.content = content
    }
    
    private var shouldUseVertical: Bool {
        hSizeClass == .compact || typeSize >= .accessibility1
    }
    
    var body: some View {
        if shouldUseVertical {
            VStack(spacing: vSpacing) { content() }
        } else {
            HStack(spacing: hSpacing) { content() }
        }
    }
}

    }

struct AppShell: View {
    @Environment(\.scenePhase) private var scenePhase
    @Environment(\.colorScheme) private var colorScheme
    @Environment(\.horizontalSizeClass) private var hSizeClass
    @Environment(\.dynamicTypeSize) private var typeSize
    @Environment(\.accessibilityReduceMotion) private var reduceMotion
    
    @State private var isBlurred = false // Blur khi inactive (banking)
    
    var body: some View {
        ZStack {
            // Main content
            TabView {
                Text("Home").tabItem { Label("Home", systemImage: "house") }
                Text("Settings").tabItem { Label("Settings", systemImage: "gear") }
            }
            
            // Privacy blur overlay
            if isBlurred {
                Color(.systemBackground)
                    .ignoresSafeArea()
                    .overlay(
                        Image(systemName: "lock.shield.fill")
                            .font(.system(size: 40))
                            .foregroundStyle(.secondary)
                    )
                    .transition(reduceMotion ? .opacity : .opacity.combined(with: .scale))
            }
        }
        .animation(reduceMotion ? .none : .easeInOut(duration: 0.3), value: isBlurred)
        .onChange(of: scenePhase) { _, newPhase in
            switch newPhase {
            case .active:
                withAnimation { isBlurred = false }
            case .inactive:
                // Blur ngay khi inactive (app switcher preview)
                withAnimation { isBlurred = true }
            case .background:
                isBlurred = true
            @unknown default:
                break
            }
        }
    }
}

Đây là chủ đề rất hay — `@Environment` là hệ thống dependency injection built-in khổng lồ của SwiftUI với hàng chục properties. Mình sẽ giải thích `scenePhase` chi tiết rồi đi qua toàn bộ các Environment values quan trọng nhất theo từng nhóm.Mình đã tổng hợp toàn bộ hệ thống `@Environment` values trong 13 phần với bảng tổng hợp 30+ properties. Đây là những điểm cốt lõi:

**`scenePhase` (Phần 1)** là Environment value quan trọng nhất cho app lifecycle. Ba trạng thái `.active` → `.inactive` → `.background` tương ứng 3 hành động: refresh data, pause operations, save state. Điểm hay: đặt ở **App level** nhận phase cho tất cả scenes (global save), đặt ở **View level** nhận cho scene cụ thể (pause video trong 1 tab).

**Top 10 Environment values dùng nhiều nhất** chia thành 4 nhóm:

**Actions (dùng hàng ngày):**
- `\.dismiss` — đóng sheet/navigation, thay thế `presentationMode` deprecated
- `\.openURL` — mở Safari, deep links, Settings app, có completion handler

**Display & Layout (responsive design):**
- `\.colorScheme` — adaptive UI cho Light/Dark mode
- `\.horizontalSizeClass` — Compact (iPhone) vs Regular (iPad) → chuyển VStack↔HStack
- `\.dynamicTypeSize` — accessibility text sizes, `.isAccessibilitySize` để detect large text
- `\.pixelLength` — 1 pixel thật (1/displayScale) cho hairline borders

**Accessibility (Apple yêu cầu):**
- `\.accessibilityReduceMotion` — **BẮT BUỘC kiểm tra** cho mọi animation phức tạp, thay spring bằng opacity/none
- `\.accessibilityReduceTransparency` — opaque background thay material/blur
- `\.accessibilityDifferentiateWithoutColor` — thêm icon khi không chỉ dùng màu phân biệt

**Data (persistence):**
- `\.modelContext` — SwiftData CRUD operations
- `\.redactionReasons` — skeleton loading + privacy cho widgets

**Production pattern hay nhất (Phần 11e — App Shell)**: Banking app blur content khi `scenePhase == .inactive` (app switcher preview không thấy số dư), kết hợp `reduceMotion` cho animation, `colorScheme` cho adaptive UI — tất cả trong 1 view sử dụng 5 Environment values cùng lúc.
